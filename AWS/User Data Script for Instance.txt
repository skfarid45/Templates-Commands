Instance 1 :
============

#!/bin/bash
# Update and install Apache (httpd) on Amazon Linux
sudo yum update -y
sudo yum install -y httpd

# Start and enable Apache
sudo systemctl start httpd
sudo systemctl enable httpd

# Create index.html with server details
cat <<EOF > /var/www/html/index.html
<h1>Server Details</h1>
<p><strong>Hostname:</strong> $(hostname)</p>
<p><strong>IP Address:</strong> $(hostname -I | awk '{print $1}')</p>
EOF

# Restart Apache
sudo systemctl restart httpd

#!/bin/bash
# Update and install Apache (httpd) on Amazon Linux
sudo yum update -y
sudo yum install -y httpd

# Start and enable Apache
sudo systemctl start httpd
sudo systemctl enable httpd

# Get server details
HOSTNAME=$(hostname)
IPADDR=$(hostname -I | awk '{print $1}')
DATE=$(date '+%A, %d %B %Y %T %Z')

# Create a stylish index.html for Server 2
cat <<EOF > /var/www/html/index.html
<!DOCTYPE html>
<html lang="en">
<head>
<meta charset="UTF-8">
<title>Server 2 Details</title>
<style>
    body {
        font-family: Arial, sans-serif;
        background: linear-gradient(to right, #6a11cb, #2575fc);
        color: #fff;
        display: flex;
        flex-direction: column;
        align-items: center;
        justify-content: center;
        height: 100vh;
        margin: 0;
        text-align: center;
    }
    h1 {
        font-size: 3em;
        margin-bottom: 0.5em;
        text-shadow: 2px 2px 4px #000;
    }
    p {
        font-size: 1.5em;
        margin: 0.3em 0;
        background: rgba(0,0,0,0.3);
        padding: 10px 20px;
        border-radius: 10px;
    }
    footer {
        margin-top: 2em;
        font-size: 1em;
        opacity: 0.8;
    }
</style>
</head>
<body>
<h1>Server 2 Dashboard</h1>
<p><strong>Hostname:</strong> $HOSTNAME</p>
<p><strong>IP Address:</strong> $IPADDR</p>
<p><strong>Date:</strong> $DATE</p>
<footer>Powered by AWS EC2 & Apache</footer>
</body>
</html>
EOF

# Restart Apache to apply changes
sudo systemctl restart httpd
